create a new repository on the command line
git init     //初始化
git add *    //添加所有文件
git commit -m "first commit"    //commit提交命名
git remote add origin git@github.com:xiaozou19921123/Test2.git  //连接仓库
git push -u origin master  //和GitHub建立连接
git push  //同步

push an existing repository from the command line
git remote add origin git@github.com:xiaozou19921123/Test2.git
git push -u origin master
